sl C:\PED\SCRIPTS\

.\addobjectstomultiplegroups.ps1 -CSV C:\PED\CSVs\oodba_temp_admins_grp_membership.csv

DISABLE old EA/DA/Tier0 admin user accounts (to be replaced by new Tier 0 accounts later)

import-csv C:\PED\CSVs\oodba_will_be_permanently_disabled.csv | %{set-aduser $_.samaccountname -Enabled:$false -SmartcardLogonRequired:$true}

import-csv C:\PED\CSVs\oodba_will_be_recreated.csv | %{set-aduser $_.samaccountname -Enabled:$false -SmartcardLogonRequired:$true}

Restart All DCs according to Reboot plan (start with PDC)

sl C:\PED\SCRIPTS\

.\RemoteServerRestart.ps1 -CSV C:\PED\CSVs\oodba_reboot_plan.csv

Create and issue new Tier 0 admin accounts & passwords to administrators

sl C:\PED\SCRIPTS\

.\BulkADUserCreationToSpecificOU.ps1 -csv C:\PED\CSVs\oodba_new_admins_to_be_created.csv

Copy-Item C:\PED\SCRIPTS\new_accounts.csv C:\ped\CSVs\oodba_recreated_admins.csv

Create brand new accounts

sl C:\PED\SCRIPTS\

.\BulkADUserCreationToSpecificOU.ps1 -csv C:\PED\CSVs\oodba_brand_new_admins_to_be_created.csv

Copy-Item C:\PED\SCRIPTS\new_accounts.csv C:\ped\CSVs\oodba_brand_new_admins.csv

Configure Tier 0 admins as "account sensitive and can't be delegated"

import-csv C:\ped\CSVs\oodba_brand_new_admins.csv | %{set-aduser $_.samaccountname -AccountNotDelegated:$true}

import-csv C:\ped\CSVs\oodba_recreated_admins.csv | %{set-aduser $_.samaccountname -AccountNotDelegated:$true}

Grant EA/DA admin rights to permanent admin accounts

sl C:\PED\SCRIPTS\

.\addobjectstomultiplegroups.ps1 -CSV C:\PED\CSVs\oodba_new_admins_grp_membership.csv

Remove all temporary accounts and permissions

sl C:\PED\SCRIPTS\

.\deleteusers.ps1 -csv C:\PED\CSVs\oodba_temp_admins.csv
